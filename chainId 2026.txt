6. INFRA (docker-compose.yml)
text
version: '3.8'
services:
  besu:
    image: hyperledger/besu:latest
    ports:
      - "8545:8545"  # RPC
      - "30303:30303" # P2P
    volumes:
      - ./genesis.json:/config/genesis.json
    command: >
      --data-path=/data 
      --genesis-file=/config/genesis.json
      --rpc-http-enabled --rpc-http-api=ETH,NET,QBFT
      --host-allowlist=* --p2p-port=30303

  explorer:
    image: blockscout/blockscout:latest
    ports:
      - "4000:80"
    environment:
      - BLOCKSCOUT_PROTOCOL=http
      - BLOCKSCOUT_HOST=localhost
      - ETH_NODE_URL=http://besu:8545
